using ll=long long;
using pii=pair<int,int>;
using pll=pair<ll,ll>;
using vi=vector<int>;
using vll=vector<ll>;
const int nax=1007;

int n, m;

int ojf[nax];
int roz[nax];

vi wek;

int fin(int v)
{
	if (v!=ojf[v])
		ojf[v]=fin(ojf[v]);
	return ojf[v];
}

set<vi> bylo[nax];

int rek(int v, vi chce)
{
	if (v==(int)wek.size())
		return 1;
	sort(chce.begin(), chce.end());
	if (bylo[v].count(chce))
		return 0;
	bylo[v].insert(chce);
	int x=wek[v];
	for (int i=0; i<(int)chce.size(); i++)
	{
		if (chce[i]>=x && (i==0 || chce[i]!=chce[i-1]))
		{
			vi now=chce;
			now[i]-=x;
			if (rek(v+1, now))
				return 1;
		}
	}
	return 0;
}

int check(int k)
{
	for (int i=0; i<=(int)wek.size(); i++)
		bylo[i].clear();
	return rek(0, vi(k, n/k));
}

void test()
{
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++)
	{
		ojf[i]=i;
		roz[i]=0;
	}
	for (int i=1; i<=m; i++)
	{
		int a, b;
		scanf("%d%d", &a, &b);
		ojf[fin(a)]=fin(b);
	}
	for (int i=1; i<=n; i++)
		roz[fin(i)]++;
	wek.clear();
	for (int i=1; i<=n; i++)
		if (fin(i)==i)
			wek.push_back(roz[i]);
	//~ debug() << imie(wek);
	sort(wek.begin(), wek.end());
	reverse(wek.begin(), wek.end());
	for (int k=1; k<=n; k++)
		if (!(n%k) && check(k))
				printf("%d ", k);
	printf("\n");
}

int main()
{
	int t;
	scanf("%d", &t);
	for (int i=1; i<=t; i++)
	{
		printf("Case #%d: ", i);
		test();
		fflush(stdout);
	}
	return 0;
}
