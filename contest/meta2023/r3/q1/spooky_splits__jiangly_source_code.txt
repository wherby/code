#include <bits/stdc++.h>

using i64 = long long;
struct DSU {
    std::vector<int> f, siz;
    
    DSU() {}
    DSU(int n) {
        init(n);
    }
    
    void init(int n) {
        f.resize(n);
        std::iota(f.begin(), f.end(), 0);
        siz.assign(n, 1);
    }
    
    int find(int x) {
        while (x != f[x]) {
            x = f[x] = f[f[x]];
        }
        return x;
    }
    
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    
    bool merge(int x, int y) {
        x = find(x);
        y = find(y);
        if (x == y) {
            return false;
        }
        siz[x] += siz[y];
        f[y] = x;
        return true;
    }
    
    int size(int x) {
        return siz[find(x)];
    }
};
void solve() {
    int N;
    std::cin >> N;
    
    std::cerr << "N : " << N << "\n";
    
    DSU dsu(N);
    
    int M;
    std::cin >> M;
    
    for (int i = 0; i < M; i++) {
        int A, B;
        std::cin >> A >> B;
        A--, B--;
        dsu.merge(A, B);
    }
    
    std::vector<std::vector<int>> S;
    S.push_back({});
    for (int i = 0; i < N; i++) {
        if (dsu.find(i) == i) {
            int x = dsu.size(i);
            std::vector<std::vector<int>> T;
            for (auto a : S) {
                for (int i = 0; i < a.size(); i++) {
                    auto b = a;
                    b[i] += x;
                    std::sort(b.begin(), b.end());
                    if (b.back() * b.size() <= N) {
                        T.push_back(b);
                    }
                }
                a.push_back(x);
                std::sort(a.begin(), a.end());
                if (a.back() * a.size() <= N) {
                    T.push_back(a);
                }
            }
            std::sort(T.begin(), T.end());
            T.erase(std::unique(T.begin(), T.end()), T.end());
            S = std::move(T);
        }
    }
    std::vector<int> ans;
    for (int K = 1; K <= N; K++) {
        if (N % K == 0 && std::binary_search(S.begin(), S.end(), std::vector(K, N / K))) {
            ans.push_back(K);
        }
    }
    for (auto x : ans) {
        std::cout << x << " \n"[x == ans.back()];
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int T;
    std::cin >> T;
    
    for (int i = 1; i <= T; i++) {
        std::cerr << "Case #" << i << ": ";
        std::cout << "Case #" << i << ": ";
        solve();
    }
    
    return 0;
}
