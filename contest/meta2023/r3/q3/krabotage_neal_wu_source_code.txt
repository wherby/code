template<class Fun> class y_combinator_result {
    Fun fun_;
public:
    template<class T> explicit y_combinator_result(T &&fun): fun_(std::forward<T>(fun)) {}
    template<class ...Args> decltype(auto) operator()(Args &&...args) { return fun_(std::ref(*this), std::forward<Args>(args)...); }
};
template<class Fun> decltype(auto) y_combinator(Fun &&fun) { return y_combinator_result<std::decay_t<Fun>>(std::forward<Fun>(fun)); }

void dbg_out() { cerr << endl; }
template<typename Head, typename... Tail> void dbg_out(Head H, Tail... T) { cerr << ' ' << H; dbg_out(T...); }
#ifdef NEAL_DEBUG
#define dbg(...) cerr << '[' << __FILE__ << ':' << __LINE__ << "] (" << #__VA_ARGS__ << "):", dbg_out(__VA_ARGS__)
#else
#define dbg(...)
#endif


const int64_t INF64 = int64_t(2e18) + 5;

void run_case(int test_case) {
    int R, C;
    cin >> R >> C;
    vector A(R, vector<int64_t>(C));

    for (auto &row : A)
        for (auto &a : row)
            cin >> a;

    vector dp_suffix(R + 1, vector<int64_t>(C + 1, 0));

    for (int r = R - 1; r >= 0; r--)
        for (int c = C - 1; c >= 0; c--)
            dp_suffix[r][c] = A[r][c] + max(dp_suffix[r + 1][c], dp_suffix[r][c + 1]);

    vector dp_prefix(R + 1, vector<int64_t>(C + 1, 0));

    for (int r = 0; r < R; r++)
        for (int c = 0; c < C; c++)
            dp_prefix[r + 1][c + 1] = A[r][c] + max(dp_prefix[r][c + 1], dp_prefix[r + 1][c]);

    vector row_sum(R, vector<int64_t>(C + 1, 0));
    vector col_sum(C, vector<int64_t>(R + 1, 0));

    for (int r = 0; r < R; r++)
        for (int c = 0; c < C; c++) {
            row_sum[r][c + 1] = row_sum[r][c] + A[r][c];
            col_sum[c][r + 1] = col_sum[c][r] + A[r][c];
        }

    vector horizontal(R, vector(C, vector<int64_t>(C, 0)));

    for (int r = 0; r < R; r++)
        for (int c = 0; c < C; c++)
            for (int d = c + 1; d < C; d++) {
                int64_t &horiz = horizontal[r][c][d];
                int64_t y_suffix_max = 0;

                for (int y = max(d - 1, c + 1); y <= d + 1; y++)
                    y_suffix_max = max(y_suffix_max, dp_suffix[y > d ? r : r + 1][y]);

                for (int x = d - 1; x >= c - 1; x--) {
                    if (x >= c + 1 && x <= d + 1) {
                        int y = x;
                        y_suffix_max = max(y_suffix_max, dp_suffix[y > d ? r : r + 1][y]);
                    }

                    horiz = max(horiz, dp_prefix[(x < c ? r : r - 1) + 1][x + 1] + y_suffix_max);
                }

                // for (int x = c - 1; x < d; x++)
                //     for (int y = max(x, c + 1); y <= d + 1; y++)
                //         horiz = max(horiz, dp_prefix[(x < c ? r : r - 1) + 1][x + 1] + dp_suffix[y > d ? r : r + 1][y]);

                dbg(r, c, d, horiz);
            }

    vector vertical(C, vector(R, vector<int64_t>(R, 0)));

    for (int c = 0; c < C; c++)
        for (int r = 0; r < R; r++)
            for (int s = r + 1; s < R; s++) {
                int64_t &vert = vertical[c][r][s];
                int64_t y_suffix_max = 0;

                for (int y = max(s - 1, r + 1); y <= s + 1; y++)
                    y_suffix_max = max(y_suffix_max, dp_suffix[y][y > s ? c : c + 1]);

                for (int x = s - 1; x >= r - 1; x--) {
                    if (x >= r + 1 && x <= s + 1) {
                        int y = x;
                        y_suffix_max = max(y_suffix_max, dp_suffix[y][y > s ? c : c + 1]);
                    }

                    vert = max(vert, dp_prefix[x + 1][(x < r ? c : c - 1) + 1] + y_suffix_max);
                }

                // for (int x = r - 1; x < s; x++)
                //     for (int y = max(x, r + 1); y <= s + 1; y++)
                //         vert = max(vert, dp_prefix[x + 1][(x < r ? c : c - 1) + 1] + dp_suffix[y][y > s ? c : c + 1]);

                dbg(c, r, s, vert);
            }

    vector dp(R, vector(C, array<int64_t, 2>({INF64, INF64})));
    dp[0][C - 1] = {0, 0};

    for (int r = 0; r < R; r++)
        for (int c = C - 1; c >= 0; c--)
            for (int h = 0; h <= 1; h++) {
                if (h == 0) {
                    for (int x = 0; x < r; x++)
                        dp[r][c][h] = min(dp[r][c][h], max(dp[x][c][!h], vertical[c][x][r]));
                } else {
                    for (int x = c + 1; x < C; x++)
                        dp[r][c][h] = min(dp[r][c][h], max(dp[r][x][!h], horizontal[r][c][x]));
                }

                dbg(r, c, h, dp[r][c][h]);
            }

    cout << "Case #" << test_case << ": " << min(dp[R - 1][0][0], dp[R - 1][0][1]) << '\n';
}

int main() {
    int tests;
    cin >> tests;

    for (int tc = 1; tc <= tests; tc++) {
        run_case(tc);
        cout << flush;
    }
}
