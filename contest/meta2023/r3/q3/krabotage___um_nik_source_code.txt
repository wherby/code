ll floor_div(ll x, ll y) {
	assert(y != 0);
	if (y < 0) {
		y = -y;
		x = -x;
	}
	if (x >= 0) return x / y;
	return (x + 1) / y - 1;
}
ll ceil_div(ll x, ll y) {
	assert(y != 0);
	if (y < 0) {
		y = -y;
		x = -x;
	}
	if (x <= 0) return x / y;
	return (x - 1) / y + 1;
}

const ll INF = (ll)1e15;
const int N = 333;
int n, m;
ll a[N][N];
ll A[N][N], B[N][N];
ll dp[N][N][2];

void solve() {
	scanf("%d%d", &n, &m);
	for (int x = 0; x <= n + 3; x++)
		for (int y = 0; y <= m + 3; y++)
			a[x][y] = A[x][y] = B[x][y] = 0;
	for (int x = 1; x <= n; x++)
		for (int y = 1; y <= m; y++)
			scanf("%lld", &a[x][y]);
	for (int x = 1; x <= n; x++)
		for (int y = 1; y <= m; y++)
			A[x][y] = a[x][y] + max(A[x - 1][y], A[x][y - 1]);
	for (int x = n; x >= 1; x--)
		for (int y = m; y >= 1; y--)
			B[x][y] = a[x][y] + max(B[x + 1][y], B[x][y + 1]);
	for (int x = 1; x <= n; x++)
		for (int y = 1; y <= m; y++)
			dp[x][y][0] = dp[x][y][1] = INF;
	dp[1][m][0] = dp[1][m][1] = 0;
	for (int x = 1; x <= n; x++)
		for (int y = m; y >= 1; y--) {
			{
				ll cur = dp[x][y][0];
				for (int z = y - 1; z >= 1; z--) {
					cur = max(cur, A[x - 1][z] + B[x + 1][z + 1]);
					dp[x][z][1] = min(dp[x][z][1], max(cur, A[x][z - 1] + B[x][y + 1]));
					cur = max(cur, A[x - 1][z] + B[x + 1][z]);
				}
			}
			{
				ll cur = dp[x][y][1];
				for (int z = x + 1; z <= n; z++) {
					cur = max(cur, A[z - 1][y - 1] + B[z][y + 1]);
					dp[z][y][0] = min(dp[z][y][0], max(cur, A[x - 1][y] + B[z + 1][y]));
					cur = max(cur, A[z][y - 1] + B[z][y + 1]);
				}
			}
		}
	printf("%lld\n", min(dp[n][1][0], dp[n][1][1]));
}

int main() {
	startTime = clock();
