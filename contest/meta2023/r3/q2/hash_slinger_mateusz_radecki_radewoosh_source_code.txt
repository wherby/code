const int nax=9007;
const int d=(1<<13);

int n, m;

int tab[nax];

int dp[d];
int juz[d];

int naj[nax][d];

void test()
{
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++)
		scanf("%d", &tab[i]);
	for (int i=0; i<d; i++)
	{
		dp[i]=n+1;
		juz[i]=0;
	}
	dp[0]=0;
	for (int i=0; i<=n+3; i++)
		for (int j=0; j<d; j++)
			naj[i][j]=n+1;
	for (int i=1; i<=n; i++)
	{
		int s=0;
		for (int j=i; j<=n; j++)
		{
			s=(s+tab[j])%m;
			naj[i][s]=min(naj[i][s], j);
		}
	}
	for (int i=n; i; i--)
		for (int j=0; j<d; j++)
			naj[i][j]=min(naj[i][j], naj[i+1][j]);
	int wyn=0;
	for (int h=0; h<d; h++)
	{
		int x=-1;
		for (int i=0; i<d; i++)
			if (!juz[i] && (x==-1 || dp[i]<dp[x]))
				x=i;
		juz[x]=1;
		if (dp[x]<=n)
			wyn++;
		else
			break;
		int g=dp[x];
		for (int j=0; j<d; j++)
			dp[x^j]=min(dp[x^j], naj[g+1][j]);
	}
	printf("%d\n", wyn);
}

int main()
{
	int t;
	scanf("%d", &t);
	for (int i=1; i<=t; i++)
	{
		printf("Case #%d: ", i);
		test();
		fflush(stdout);
	}
	return 0;
}
